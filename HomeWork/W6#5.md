    IN in, load;
    OUT out;

    // Put your code here:

    // in 接到b (詳情參見Mux)
    Mux(b=in,a=o,sel=load,out=Muxout);     //不允許out出去還拉回來，用個Ｏ做中間替代

}</code></pre>
